EVALUATE
SELECTCOLUMNS(
	FILTER(
		INFO.MEASURES(),
		[IsHidden] = FALSE() &&
		[IsSimpleMeasure] = FALSE()
	),
	"MEASURE_ID", [ID],
	"TABLE_ID", [TableID],
	"MEASURE_NAME", [Name],
	"DESCRIPTIOM", [Description],
	"DATA_TYPE_ID", [DataType],
	"DATA_TYPE", SWITCH(
					[DataType],
					8, "Decimal",
					6, "Integer",
					20, "Percentage",
					BLANK()
				),
	"DAX_EXPRESSION", [Expression],
	"IS_HIDDEN", [IsHidden],
	"MODIFIED_DATE", [ModifiedTime],
	"DISPLAY_FOLDER", [DisplayFolder]
)
